import { NextRequest, NextResponse } from "next/server";
import { run__FUNC_PREFIX__Analysis } from "../../_lib/__NODE_ID__";

export const runtime = "nodejs";

export async function POST(req: NextRequest) {
  try {
    const body = await req.json();

    const result = run__FUNC_PREFIX__Analysis({
      user_query: typeof body?.user_query === "string" ? body.user_query : "",
    });

    return NextResponse.json({
      ok: true,
      node_id: "__NODE_ID__",
      result,
    });
  } catch (error) {
    const message =
      error instanceof Error ? error.message : "Unknown __NODE_ID__ route error";

    return NextResponse.json(
      {
        ok: false,
        node_id: "__NODE_ID__",
        error: message,
      },
      { status: 500 }
    );
  }
}
